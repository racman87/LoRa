%--- Console output -----------------------------------------------------------------
% Console command in paragraph
\newcommand{\cmdconsole}{
	\mintinline{vctreestatus}
}
% Console output
\newminted[codeconsole]{vctreestatus}{
	frame=lines,
	framesep=2mm,
	baselinestretch=1.2,
	breaklines=true,
	fontsize=\footnotesize
}

%--- Bash code ----------------------------------------------------------------------
% bash code in paragraph
\newcommand{\cmdbash}{
	\mintinline{bash}
}

% Bash code in document
\newminted[codebash]{bash}{
	frame=lines,
	framesep=2mm,
	baselinestretch=1.2,
	breaklines=true,
	fontsize=\footnotesize
}

% Bash code with line number in document
\newminted[codebashlnb]{bash}{
	frame=lines,
	framesep=2mm,
	baselinestretch=1.2,
	fontsize=\footnotesize,
	linenos,
	breaklines=true,
	stepnumber=5
}

% Bash code from file
\newcommand{\icodebash}[1]{
	\inputminted[%
		frame=lines,
		framesep=2mm,
		baselinestretch=1.2,
		fontsize=\footnotesize,
		linenos,
		breaklines=true,
		stepnumber=5]{bash}{#1}
}
